\noindent
For each of the following series, determine whether or not the series
converges.  Be sure to mention which test you are using, and to
describe how you applied the test.

\vfill

\begin{multicols}{2}

\begin{problem}[1]
$$
\sum_{n=1}^\infty \frac{1}{n^2}
$$
\end{problem}

\vfill

\begin{problem}[1]
$$
\sum_{n=1}^\infty \frac{1}{n \cdot \sqrt{n}}
$$
\end{problem}

\vfill

\begin{problem}[1]
$$
\sum_{n=1}^\infty \frac{1}{n \cdot (n+1)}.
$$
\end{problem}

\vfill

\begin{problem}[2]
$$
\sum_{n=1}^\infty \frac{1}{2 + \sin^2 n}.
$$
\end{problem}

\vfill

\begin{problem}[2]
$$
\sum_{n=3}^\infty \frac{2n + 17}{n^2 - 5}
$$
\end{problem}

\vfill

\begin{problem}[2]
$$
\sum_{n=1}^\infty \frac{n + 2}{2n + 3}
$$
\end{problem}

\vfill

\begin{problem}[2]
$$
\sum_{n=1}^\infty \frac{4^n + n}{n!}
$$
\end{problem}

\vfill

\begin{problem}[2]
$$
\sum_{n=1}^\infty \frac{n}{1 + n \cdot 2^n}
$$
\end{problem}

\vfill

\begin{problem}[2]
$$
\sum_{n=1}^\infty \frac{n!}{n^{15}}
$$
\end{problem}
