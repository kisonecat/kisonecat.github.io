\begin{problem}[2]
Define $F : \R \to \R$ by the rule
$$
F(x) = (x+17)^3.
$$
and find functions $f$ and $g$ so that $F = f \circ g$.
\end{problem}

\begin{problem}[2]
Define a function $f : \R \to \R$ by
$$
f(x) = \begin{cases}
x & \mbox{ if $x < 0$, } \\
\sqrt{2} & \mbox{ if $x = 0$, } \\
x^2 & \mbox{ if $x > 0$. }
       \end{cases}
$$
What is $\lim_{x \to 0} f(x)$?
\end{problem}

\begin{problem}[3]
Evaluate this limit:
$$
\lim_{x \to 1} \frac{x^3 - 1}{x-1}.
$$
\end{problem}

\begin{problem}[3]
Give an $\epsilon$-$\delta$ proof of the fact that
$$
\lim_{x \to 1} 3x + 2 = 5.
$$
\end{problem}
