I give you a selection of problems about trigonometry.

%This quiz is worth 14 points---12 points are for ``correctness'' of
%answers, but an additional 2 points will be given out for
%``style''---awarded to answers which are clearly written, easy to
%follow, etc.

\begin{problem}[2]
Prove that
$$
(1 + \cos \theta)(1 - \cos \theta) = \sin^2 \theta
$$
is an identity.
\end{problem}

\begin{problem}[2]
A bicycle wheel, 27~inches in diameter, has rotated through an angle
of $60^\circ$.  How far has the bicycle moved?
\end{problem}

\begin{problem}[2]
Find the limit, if it exists:
$$
\lim_{\theta \to 0} \frac{\sin \theta - \cos \theta}{3\theta + 2 + \cos \theta}.
$$
\end{problem}

\begin{problem}[2]
Find the limit, if it exists:
$$
\lim_{x \to 0} \frac{\tan x}{x}.
$$
\end{problem}

\begin{problem}[3]
Find the limit, if it exists:
$$
\lim_{x \to 0} \frac{\sin (4x)}{5x}.
$$
\end{problem}

\begin{problem}[3]
Find the limit, if it exists:
$$
\lim_{x \to 0} \frac{\sin^2 x}{x}.
$$
\end{problem}
