\begin{problem}[2]
What is the area of the region above $y=0$ and below $y = 1 - x^4$?
\end{problem}

\begin{problem}[2]
What is the area of the region above $y = x^2$ and below $y =
\sqrt{x}$?
\end{problem}

\begin{problem}[2]
What is the area of the region bounded on the left by $x = 2y^2 - 4$
and on the right by $x = y^2$?
\end{problem}

\begin{problem}[2]
Let $R$ be the region bounded by $y = 0$, $y = x^2 + 1$, $x = -1$, $x
= 1$.  Rotate the region $R$ around the $x$-axis, and compute the
volume of the resulting solid.
\end{problem}

\begin{problem}[3]
Define the following:
$$
R_a = \{ (x,y) \in \R^2 : 0 \leq y \leq \sqrt{a^2-x^2}, -a \leq x \leq a \},
$$ so $R_a$ is a semicircle of radius $a$.  Rotate $R_a$ around the
$x$-axis, and (using the method of disks) compute the volume of the
resulting solid---which is a sphere of radius $a$.  Compare your
answer to the formula you know.
\end{problem}

\begin{problem}[3]
Let $M$ be a very large number.  Consider the region $R$ given by $1
\leq x \leq M$, and $0 \leq y \leq 1/x$.  Sketch the region.  Rotate
$R$ around the $x$-axis, and compute the volume of the resulting solid
in terms of $M$.  What happens if you take the limit as $M$ tends to
infinity?
\end{problem}
